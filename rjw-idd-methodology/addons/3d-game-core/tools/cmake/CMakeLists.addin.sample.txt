# RJW-IDD 3D Game Core Add-in CMake Sample
# Usage: include this snippet conditionally when the feature flag is enabled.

include_guard(GLOBAL)

# Expect the parent project to define RJW3D_FEATURES_JSON or read features.yml via Python.
if(NOT DEFINED RJW3D_ADDIN_ENABLED)
  message(FATAL_ERROR "RJW3D_ADDIN_ENABLED is not defined. Gate inclusion through feature toggle scripts.")
endif()

if(RJW3D_ADDIN_ENABLED)
  list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/modules")
  include(RJWPerfGate3D)
  include(RJWAssetLint3D)
endif()
